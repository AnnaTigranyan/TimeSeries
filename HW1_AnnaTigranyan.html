Problem 3 (35 points)


Create a function in R which generate AR(2) process.

ar2_generator <- function(m,a,b,n) {

  x <- vector("numeric", n)
  set.seed(123)
  w <- rnorm(n)
  x[1] = w[1]
  
  
  for  (t in 3:n){
    x[t] = m + a*x[t-1] + b*x[t-2] + w[t]
  } 
  
  par(mar = c(1,1,1,1))
  acf(x)
  
  p1<- autoplot(acf(x))

  
  p2 <- autoplot(pacf(x)) + ylab("PACF")

  
  p3 <- ggplot(as.xts(x, order.by = ymd("2022-10-19")-0:(n-1))) + 
    geom_line(aes(x=Index, y=x)) + ylab(" ") + xlab("Time")

  
  p4 <- ggplot(as.xts(x, order.by = ymd("2022-10-19")-0:(n-1))) + 
    geom_point(aes(x=x, y=Lag(x,1))) + geom_smooth(aes(x=x, y=Lag(x,1)),method="lm", se =FALSE)
  
  grid.arrange(p1,p2,p3,p4)
  
  A <- adf.test(x)
  A
  
}

ar2_generator(1,0.6,0.3,300)
